public static void main(String[] args) {
     int a = Integer.parseInt(JOptionPane.showInputDialog("Digite a quantidade de alunos: "));
     String nome[] = new String[a]; //Vetor para os nomes dos alunos.
     Float notas[] =  new Float[a]; // Vetor para as notas dos alunos.
     for(int i = 0; i < nome.length; i++){ //Laço para ler dados digitados pelo usuario;
         nome[i] = JOptionPane.showInputDialog("Digite o nome do "+(i+1)+" aluno: ");
         notas [i] = Float.parseFloat(JOptionPane.showInputDialog("Digite a nota do aluno "+nome[i]+": "));
     }
     for(int b = 0; b < nome.length; b++){ //Laço para imprimir os vetores;
         System.out.println("Aluno: "+nome[b]);
         System.out.println("Nota: "+notas[b]);
         System.out.println("---------------------------");
     }
    }
