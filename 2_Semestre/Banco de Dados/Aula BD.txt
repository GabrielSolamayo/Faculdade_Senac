create table cliente (
       cod_cli int not null primary key,
       nome_cli varchar(50) not null,
       endereco varchar(50) not null,
       cidade varchar(20) null,
       cep char(9) null,
       uf char(2)null
);

desc produto;
create table produto 
( 
 cod_prod int not null auto_increment, 
 uni_prod char(03) not null, 
 desc_prod varchar(20) not null,
 val_unit numeric(9,2) not null,
 primary key (cod_prod) );

desc vendedor;
create table Vendedor (
     cod_vend int not null primary key auto_increment,
     nome_vend varchar(50) not null,
     sal_fixo decimal(9,2),
     faixa_comiss char(1)
);

desc pedido;
create table pedido(
        num_pedido int not null primary key auto_increment,
        prazo int not null,
        fk_cod_vend int not null,
        fk_cod_cli int not null,
        foreign key (fk_cod_vend) references Vendedor(cod_vend),
        foreign key (fk_cod_cli) references Cliente(cod_cli)
);

CREATE TABLE item_pedido(
  no_ped int(11) not null,
  cd_prod int(11) not null,
  qtd_ped float not null
);
